304 static int GetGlyphInfo( const GLCD_FontDef* Font, const char c, int* OutGlyphWidth, uint16_t* OutGlyphOffset ) {
305  uint16_t GlyphOffset = 0;
306  int GlyphWidth = 0;
307  int GlyphYBytes = 0;
308  int GlyphSize = 0;
309  int i = 0;
310 
311  if ( Font == NULL || c < Font->FirstChar || c > Font->LastChar ) {
312  return 0;
313  }
314 
315  GlyphYBytes = ( Font->FontHeightPxRoundedUp / 8 );
316 
317  if ( Font->IsStream == true ) {
318  for ( i = 0; i < ( c - Font->FirstChar ); i++ ) {
319  GlyphWidth = GlyphReadByte( Font, GlyphOffset );
320  GlyphSize = ( ( GlyphWidth * GlyphYBytes ) ) + 1;
321  GlyphOffset+= GlyphSize;
322  }
323 
324  NullCheckAndAssign( OutGlyphWidth, GlyphReadByte( Font, GlyphOffset++ ) );
325  NullCheckAndAssign( OutGlyphOffset, GlyphOffset );
326  } else {
327  GlyphWidth = Font->FontWidthPx;
328  GlyphSize = ( GlyphWidth * GlyphYBytes ) + 1;
329 
330  GlyphOffset = ( GlyphSize * ( c - Font->FirstChar ) );
331 
332  if ( Font->IsExternal ) {
333  GlyphOffset+= ( uint16_t ) Font->Data;
334  }
335 
336  NullCheckAndAssign( OutGlyphWidth, GlyphReadByte( Font, GlyphOffset ) );
337  NullCheckAndAssign( OutGlyphOffset, ( ++GlyphOffset ) );
338  }
339 
340  return 1;
341 }
